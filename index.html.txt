    <script>
        const noBtn = document.querySelector('.btn-no');
        const yesBtn = document.querySelector('.btn-yes');
        const questionScreen = document.getElementById('question-screen');
        const successScreen = document.getElementById('success-screen');
        const container = document.querySelector('.container');
        const containerRect = container.getBoundingClientRect();

        // 1. Logic to make the "No" button run away
        function moveButton() {
            // Get the dimensions of the container to keep button inside (or close to it)
            const width = window.innerWidth;
            const height = window.innerHeight;
            
            // Calculate a random position
            // We ensure it stays somewhat visible but hard to click
            const newX = Math.random() * (width - 100); 
            const newY = Math.random() * (height - 50);

            // Apply the new position
            noBtn.style.position = "fixed"; // Switch to fixed to move anywhere on screen
            noBtn.style.left = newX + "px";
            noBtn.style.top = newY + "px";
        }

        // 2. Logic for clicking "Yes"
        function handleYes() {
            // Hide the question
            questionScreen.style.display = 'none';
            
            // Show the success
            successScreen.style.display = 'block';

            // Trigger Confetti (Celebration)
            triggerConfetti();
        }

        // 3. Confetti Effect Function
        function triggerConfetti() {
            const count = 200;
            const defaults = {
                origin: { y: 0.7 }
            };

            function fire(particleRatio, opts) {
                confetti(Object.assign({}, defaults, opts, {
                    particleCount: Math.floor(count * particleRatio)
                }));
            }

            fire(0.25, { spread: 26, startVelocity: 55 });
            fire(0.2, { spread: 60 });
            fire(0.35, { spread: 100, decay: 0.91, scalar: 0.8 });
            fire(0.1, { spread: 120, startVelocity: 25, decay: 0.92, scalar: 1.2 });
            fire(0.1, { spread: 120, startVelocity: 45 });
        }
